00045 inline arma::mat dare(const arma::mat& _A, const arma::mat& _B, const arma::mat& _Q, const arma::mat& _R) {
00046 using namespace arma;
00047 
00048   int n = _A.n_rows, m = _B.n_cols;
00049   mat Z11(n,n);
00050   mat Z12(n,n);
00051   mat Z21(n,n);
00052   mat Z22(n,n);
00053   mat temp1(m,m);
00054   mat temp2(n,n);
00055   mat temp3(m,n);
00056 
00057   //pre-compute values to save recomputation
00058   //"trans" takes Hermitian transpose of matrix,
00059   //i.e. conjugate of elements is taken during transpose operation
00060   temp1 = inv(_R);
00061   temp2 = _B * temp1 * trans(_B);
00062 
00063   //construct submatrices that will constitute Hamiltonian matrix
00064   Z11 = inv(_A);
00065   Z21 = _Q * Z11;
00066   Z12 = Z11 * temp2;
00067   Z22 = trans(_A) + Z21 * temp2;
00068 
00069   mat Z(2*n, 2*n);
00070 
00071   //construct Hamiltonian matrix in discrete case
00072   Z.submat( span(0,n-1),    span(0,n-1)   ) = Z11;
00073   Z.submat( span(n,2*n-1),  span(0,n-1)   ) = Z21;
00074   Z.submat( span(0,n-1),    span(n,2*n-1) ) = Z12;
00075   Z.submat( span(n,2*n-1),  span(n,2*n-1) ) = Z22;
00076 
00077   //data structures to store result of generalized eigenvalue computation
00078   //Col< std::complex<typename T1::pod_type> > eigval(2*n);
00079   cx_vec eigval(2*n);
00080   int n_Z = Z.n_rows;
00081   mat VL(n_Z, n_Z);
00082   mat VR(n_Z, n_Z);
00083 
00084   //solution to generalized eigenvalue problem
00085   //internally, calls Fortran's "geev" function
00086   eig_gen(eigval, VL, VR, Z);
00087 
00088   //cout << "eigen values in DARE: " << eigval << endl;
00089 
00090   mat U11(n,n);
00091   mat U21(n,n);
00092   mat tempZ(n_Z, n);
00093 
00094   int c1=0;
00095   for(int i = 0; i < n_Z; ++i) {
00096 
00097     //get eigenvalues outside unit circle
00098     if( (eigval(i).real()*eigval(i).real() +
00099     eigval(i).imag()*eigval(i).imag()  ) > 1) {
00100 
00101       tempZ.submat(span::all, span(c1,c1)) = VR.submat(span::all, span(i,i));
00102       c1++;
00103     }
00104   }
00105 
00106   //ensure that the system is stable
00107   assert(c1 == n);
00108 
00109   U11 = tempZ.submat(span(0,n-1), span(0,n-1));
00110   U21 = tempZ.submat(span(n,n_Z-1), span(0,n-1));
00111 
00112   return U21 * inv(U11);
00113 }
