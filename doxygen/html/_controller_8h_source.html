365 bool Controller<SeparatedControllerType, FilterType>::executeOneStep(const int k, const ompl::base::State *startState,
366  ompl::base::State* endState,
367  ompl::base::Cost &filteringCost,
368  bool constructionMode)
369 {
370  using namespace std;
371 
372  //HOW TO SET INITAL VALUE OF COST
373  //cost = 1 ,for time based only if time per execution is "1"
374  //cost = 0.01 , for covariance based
375  double cost = 0.01;
376 
377  ompl::base::State *internalState = si_->allocState();
378  si_->copyState(internalState, startState);
379 
380  ompl::base::State *nominalX_K = si_->allocState();
381 
382  this->Evolve(internalState, k, endState) ;
383 
384  si_->copyState(internalState, endState);
385 
386  // if the propagated state is not valid, return false (only in construction mode)
387  if(constructionMode)
388  {
389  if(!si_->checkTrueStateValidity())
390  {
391  return false;
392  }
393  }
394 
395  if(k<lss_.size())
396  nominalX_K = lss_[k].getX();
397 
398  else nominalX_K = lss_[lss_.size()-1].getX();
399 
400  arma::colvec nomXVec = nominalX_K->as<StateType>()->getArmaData();
401  arma::colvec endStateVec = endState->as<StateType>()->getArmaData();
402  arma::colvec deviation = nomXVec.subvec(0,1) - endStateVec.subvec(0,1);
403 
404  if(abs(norm(deviation,2)) > nominalTrajDeviationThreshold_)
405  {
406  return false;
407  }
408 
409  //Increment cost by:
410  //-> 0.01 for time based
411  //-> trace(Covariance) for FIRM
412  cost += arma::trace(endState->as<StateType>()->getCovariance());
413 
414  if(!constructionMode) boost::this_thread::sleep(boost::posix_time::milliseconds(20));
415 
416  //filteringCost.v = cost;
417  filteringCost = ompl::base::Cost(cost);
418 
419  si_->freeState(internalState);
420 
421  return true ;
422 }
